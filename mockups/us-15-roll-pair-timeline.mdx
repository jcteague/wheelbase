import { useState } from 'react'

# US-15: Roll Pair Display in Leg Timeline

_Display linked roll pairs (ROLL\_FROM + ROLL\_TO) as visually grouped sections in the position leg timeline with net credit/debit and sequential numbering._

## Acceptance Criteria States

- **single-roll**: One CSP roll pair shown grouped with net credit, alongside normal legs
- **multiple-rolls**: Three sequential CSP rolls with cumulative summary, followed by assignment and CC open
- **debit-roll**: A roll pair with net debit shown in amber

---

export const MONO = "'JetBrains Mono', 'Fira Mono', monospace"

export function MockBadge({ children, color }) {
  const fg = color ?? 'var(--wb-gold, #e6a817)'
  return (
    <span style={{ display: 'inline-flex', alignItems: 'center', fontSize: '0.6rem', fontWeight: 500, padding: '1px 6px', borderRadius: 10, background: fg + '18', color: fg, border: '1px solid ' + fg + '30', fontFamily: MONO }}>{children}</span>
  )
}

export function LegRow({ role, action, type, strike, exp, premium, basis, isRoll }) {
  const roleColors = {
    CSP_OPEN: '#e6a817', ROLL_FROM: '#8899aa', ROLL_TO: '#58a6ff',
    ASSIGN: '#bc8cff', CC_OPEN: '#bc8cff', CC_CLOSE: '#8899aa', EXPIRE: '#4a5a6a',
  }
  const color = roleColors[role] ?? '#8899aa'
  return (
    <div style={{
      display: 'grid', gridTemplateColumns: '110px 60px 80px 80px 90px 90px',
      padding: isRoll ? '8px 14px 8px 28px' : '10px 14px',
      borderBottom: '1px solid rgba(30,42,56,0.4)',
      background: isRoll ? 'rgba(88,166,255,0.02)' : undefined,
      fontFamily: MONO, fontSize: 11, color: 'var(--wb-text-primary, #e2eaf2)',
      alignItems: 'center',
    }}>
      <span><MockBadge color={color}>{role}</MockBadge></span>
      <span style={{ color: '#8899aa' }}>{action}</span>
      <span>{type} ${strike}</span>
      <span style={{ color: '#8899aa' }}>{exp}</span>
      <span style={{ color: premium.startsWith('+') ? 'var(--wb-green, #3fb950)' : premium.startsWith('-') ? 'var(--wb-red, #f85149)' : '#8899aa' }}>{premium}</span>
      <span style={{ textAlign: 'right', fontWeight: 600 }}>{basis}</span>
    </div>
  )
}

export function RollGroupHeader({ number, rollType, fillDate, netCredit, isCredit }) {
  const color = isCredit ? 'var(--wb-green, #3fb950)' : 'var(--wb-gold, #e6a817)'
  const bg = isCredit ? 'rgba(63,185,80,0.04)' : 'rgba(230,168,23,0.04)'
  const border = isCredit ? 'rgba(63,185,80,0.15)' : 'rgba(230,168,23,0.15)'
  return (
    <div style={{
      display: 'flex', alignItems: 'center', justifyContent: 'space-between',
      padding: '8px 14px', background: bg,
      borderTop: '2px solid ' + border, borderBottom: '1px solid rgba(30,42,56,0.4)',
      fontFamily: MONO,
    }}>
      <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
        <span style={{ fontSize: 11, fontWeight: 700, color }}>Roll #{number}</span>
        <MockBadge color={color}>{rollType}</MockBadge>
        <span style={{ fontSize: 10, color: 'var(--wb-text-muted, #4a5a6a)' }}>{fillDate}</span>
      </div>
      <span style={{ fontSize: 12, fontWeight: 700, color, fontFamily: MONO }}>
        {isCredit ? '+' : '-'}${Math.abs(parseFloat(netCredit)).toFixed(2)}/contract
      </span>
    </div>
  )
}

export function TableHeader() {
  return (
    <div style={{
      display: 'grid', gridTemplateColumns: '110px 60px 80px 80px 90px 90px',
      padding: '8px 14px',
      borderBottom: '1px solid var(--wb-border, #1e2a38)',
      fontFamily: MONO, fontSize: 9, fontWeight: 600,
      letterSpacing: '0.08em', textTransform: 'uppercase',
      color: 'var(--wb-text-muted, #4a5a6a)',
    }}>
      <span>Leg</span>
      <span>Action</span>
      <span>Contract</span>
      <span>Expiration</span>
      <span>Premium</span>
      <span style={{ textAlign: 'right' }}>Cost Basis</span>
    </div>
  )
}

export function CumulativeSummary({ totalCredits, totalDebits }) {
  const net = totalCredits - totalDebits
  const isNet = net >= 0
  const color = isNet ? 'var(--wb-green, #3fb950)' : 'var(--wb-red, #f85149)'
  return (
    <div style={{
      display: 'flex', alignItems: 'center', justifyContent: 'space-between',
      padding: '10px 14px', background: 'rgba(88,166,255,0.03)',
      borderTop: '1px solid rgba(88,166,255,0.12)', borderBottom: '1px solid var(--wb-border, #1e2a38)',
      fontFamily: MONO, fontSize: 11,
    }}>
      <span style={{ color: 'var(--wb-text-muted, #4a5a6a)' }}>Roll Summary ({totalCredits > 0 && totalDebits > 0 ? '3 rolls' : '1 roll'})</span>
      <div style={{ display: 'flex', gap: 16, alignItems: 'center' }}>
        <span style={{ color: 'var(--wb-green, #3fb950)' }}>Credits: +${totalCredits.toFixed(2)}</span>
        {totalDebits > 0 && <span style={{ color: 'var(--wb-gold, #e6a817)' }}>Debits: -${totalDebits.toFixed(2)}</span>}
        <span style={{ fontWeight: 700, color }}>Net: {isNet ? '+' : '-'}${Math.abs(net).toFixed(2)}</span>
      </div>
    </div>
  )
}
