# US-14: Roll Covered Call Form

_Roll an open covered call to a new strike and/or expiration with cost basis context and net credit/debit preview._

## Acceptance Criteria States

- **up-and-out**: Roll up and out — $185 to $190 strike, net credit $0.70/contract, cost basis shown
- **below-basis-warning**: New strike $175 is below cost basis $176.50 — amber warning displayed, still submittable
- **roll-out**: Roll out — same $185 strike, later expiration, net credit $1.30
- **success**: Roll confirmed — linked pair created, position stays CC_OPEN

---

export function NetPreview({ costToClose, newPremium, contracts }) {
  const net = parseFloat(newPremium) - parseFloat(costToClose)
  const isCredit = net >= 0
  const total = Math.abs(net) * contracts * 100
  const color = isCredit ? 'var(--wb-green, #3fb950)' : 'var(--wb-gold, #e6a817)'
  const bg = isCredit ? 'rgba(63,185,80,0.07)' : 'rgba(230,168,23,0.07)'
  const border = isCredit ? 'rgba(63,185,80,0.22)' : 'rgba(230,168,23,0.22)'
  const sign = isCredit ? '+' : '-'
  return (
    <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between', padding: '12px 14px', borderRadius: 6, background: bg, border: '1px solid ' + border }}>
      <span style={{ fontSize: 11, color: 'var(--wb-text-secondary, #8899aa)', fontFamily: MONO }}>{isCredit ? 'Net Credit' : 'Net Debit'}</span>
      <div style={{ display: 'flex', alignItems: 'baseline', gap: 8 }}>
        <span style={{ fontSize: 17, fontWeight: 700, color, fontFamily: MONO }}>{sign}${Math.abs(net).toFixed(2)}/contract</span>
        <span style={{ fontSize: 11, color, opacity: 0.7, fontFamily: MONO }}>(${total.toFixed(2)} total)</span>
      </div>
    </div>
  )
}

  const [state, setState] = useState('up-and-out')
  const pos = MOCK_POSITION
  const allStates = ['up-and-out', 'below-basis-warning', 'roll-out', 'success']
  const isSuccess = state === 'success'
  const isBelowBasis = state === 'below-basis-warning'
  const isRollOut = state === 'roll-out'

  const newStrike = isBelowBasis ? '175.00' : isRollOut ? pos.strike : '190.00'
  const costToClose = isBelowBasis ? '2.00' : isRollOut ? '1.80' : '3.50'
  const newPremium = isBelowBasis ? '2.30' : isRollOut ? '3.10' : '4.20'

  const sameStrike = newStrike === pos.strike
  const rollType = sameStrike ? 'Roll Out' : parseFloat(newStrike) > parseFloat(pos.strike) ? 'Roll Up & Out' : 'Roll Down & Out'

  const belowBasis = parseFloat(newStrike) < parseFloat(pos.basisPerShare)
  const lossPerShare = belowBasis ? (parseFloat(pos.basisPerShare) - parseFloat(newStrike)).toFixed(2) : null

  return (
    <div style={{ fontFamily: MONO, background: '#040608', minHeight: 520 }}>
      <div style={{ display: 'flex', gap: 6, padding: '10px 14px', background: '#0d1117', borderBottom: '1px solid #1e2a38', flexWrap: 'wrap' }}>
        <span style={{ fontSize: 9, color: '#4a5a6a', letterSpacing: '0.12em', textTransform: 'uppercase', alignSelf: 'center', marginRight: 4 }}>State:</span>
        {allStates.map(s => (
          <button key={s} onClick={() => setState(s)} style={{ fontSize: 10, fontFamily: MONO, padding: '3px 10px', borderRadius: 4, border: '1px solid', borderColor: s === state ? 'var(--wb-gold, #e6a817)' : '#1e2a38', background: s === state ? 'rgba(230,168,23,0.12)' : 'transparent', color: s === state ? 'var(--wb-gold, #e6a817)' : '#6a7a8a', cursor: 'pointer' }}>{s}</button>
        ))}
      </div>

      <div style={{ position: 'relative', height: 720, overflow: 'hidden', background: '#040608' }}>
        <div style={{ padding: '24px 28px', opacity: 0.25, pointerEvents: 'none', userSelect: 'none' }}>
          <div style={{ display: 'flex', alignItems: 'center', gap: 14, marginBottom: 20 }}>
            <span style={{ fontSize: 22, fontWeight: 700, color: '#e2eaf2' }}>AAPL</span>
            <MockPhaseBadge phase="CC_OPEN" />
            <span style={{ fontSize: 11, color: '#8899aa' }}>CALL $185.00 · Apr 18, 2026 · 1 contract</span>
          </div>
        </div>
        <div style={{ position: 'absolute', inset: 0, background: 'rgba(4,6,8,0.45)' }} />

        <div style={panelStyle}>
          <div style={{ padding: '20px 24px 18px', borderBottom: '1px solid ' + (isSuccess ? 'rgba(63,185,80,0.2)' : 'var(--wb-border, #1e2a38)'), display: 'flex', alignItems: 'flex-start', justifyContent: 'space-between', flexShrink: 0 }}>
            <div>
              <div style={{ fontSize: '0.65rem', fontWeight: 600, letterSpacing: '0.1em', textTransform: 'uppercase', marginBottom: 6, color: isSuccess ? 'var(--wb-green, #3fb950)' : 'var(--wb-text-secondary, #8899aa)', fontFamily: MONO }}>
                {isSuccess ? 'Roll Complete' : rollType}
              </div>
              <div style={{ fontSize: 17, fontWeight: 700, color: 'var(--wb-text-primary, #e2eaf2)', marginBottom: 4 }}>
                {isSuccess ? 'CC Rolled Successfully' : 'Roll Covered Call'}
              </div>
              <div style={{ fontSize: 11, color: 'var(--wb-text-secondary, #8899aa)' }}>
                {isSuccess ? 'CALL $185 \u2192 CALL $190 · May 16, 2026' : pos.ticker + ' CALL $' + pos.strike + ' · exp Apr 18, 2026'}
              </div>
            </div>
            <button style={{ background: 'var(--wb-bg-elevated, #0d1821)', border: '1px solid var(--wb-border, #1e2a38)', color: 'var(--wb-text-muted, #4a5a6a)', fontSize: 14, width: 28, height: 28, borderRadius: 6, cursor: 'pointer', display: 'flex', alignItems: 'center', justifyContent: 'center', flexShrink: 0 }}>{'\u00d7'}</button>
          </div>

          <div style={{ padding: '20px 24px', overflowY: 'auto', flex: 1, display: 'flex', flexDirection: 'column', gap: 16 }}>
            {isSuccess ? (
              <>
                <div style={{ background: 'linear-gradient(135deg, rgba(63,185,80,0.1), rgba(7,10,14,0.5))', border: '1px solid rgba(63,185,80,0.22)', borderRadius: 10, padding: '22px 20px', textAlign: 'center' }}>
                  <div style={{ fontSize: 9, color: 'var(--wb-green, #3fb950)', opacity: 0.75, letterSpacing: '0.18em', textTransform: 'uppercase', marginBottom: 8 }}>Roll Net Credit</div>
                  <div style={{ fontSize: 40, fontWeight: 700, color: 'var(--wb-green, #3fb950)', letterSpacing: '-0.03em', lineHeight: 1, marginBottom: 6 }}>+$0.70</div>
                  <div style={{ fontSize: 11, color: 'var(--wb-green, #3fb950)', opacity: 0.55 }}>$70.00 total · 1 contract</div>
                </div>
                <MockSectionCard>
                  <MockSummaryRow label="Roll type" value={<MockBadge color="#3fb950">Roll Up &amp; Out</MockBadge>} />
                  <MockSummaryRow label="Old leg" value="ROLL_FROM · BUY CALL $185 @ $3.50" />
                  <MockSummaryRow label="New leg" value="ROLL_TO · SELL CALL $190 @ $4.20" />

                  <MockSummaryRow label="Phase" value={<span style={{ display: 'flex', alignItems: 'center', gap: 5 }}><MockPhaseBadge phase="CC_OPEN" /><span style={{ fontSize: 9, color: 'var(--wb-text-muted, #4a5a6a)' }}>(unchanged)</span></span>} />
                  <MockSummaryRow label="Cost basis" value={<span style={{ color: 'var(--wb-green, #3fb950)' }}>$176.50 {'\u2192'} $175.80/share</span>} noBorder highlight />

              <>

                  <MockSummaryRow label="Type" value={<span style={{ display: 'flex', alignItems: 'center', gap: 6 }}><MockPhaseBadge phase="CC_OPEN" />CALL</span>} />

                  <MockSummaryRow label="Expiration" value={pos.expiration + ' (' + pos.dte + ' DTE)'} />
                  <MockSummaryRow label="Premium collected" value={<span style={{ color: 'var(--wb-green, #3fb950)' }}>+${pos.premium}/contract</span>} />
                  <MockSummaryRow label="Cost basis" value={'$' + pos.basisPerShare + '/share'} noBorder highlight />
                </MockSectionCard>

                <MockSectionCard header="New Leg">
                  <div style={{ padding: '16px 14px', display: 'flex', flexDirection: 'column', gap: 14 }}>
                    <MockField label="New Strike" hint="Raise strike to avoid call-away, or keep same for roll out">
                      <MockPriceInput value={newStrike} />
                    </MockField>
                    <MockField label="New Expiration" hint="Must be after current expiration">
                      <MockDatePicker value="2026-05-16" />
                    </MockField>
                    <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: 12 }}>
                      <MockField label="Cost to Close" hint="Buy-to-close">
                        <MockPriceInput value={costToClose} />
                      </MockField>
                      <MockField label="New Premium" hint="Sell-to-open">
                        <MockPriceInput value={newPremium} />
                      </MockField>
                    </div>
                    <MockField label="Fill Date">
                      <MockDatePicker value="2026-04-04" />
                    </MockField>
                  </div>

                  </span>
                </div>

                {/* Below-cost-basis warning */}
                {isBelowBasis && (
                  <MockAlertBox variant="error">
                    <strong style={{ fontWeight: 700, display: 'block', marginBottom: 3 }}>Strike below cost basis</strong>

                <NetPreview costToClose={costToClose} newPremium={newPremium} contracts={pos.contracts} />

                <MockAlertBox variant="warning">
                  <strong style={{ fontWeight: 700, display: 'block', marginBottom: 3 }}>This cannot be undone.</strong>

              </>

          {!isSuccess && (
            <div style={{ padding: '16px 24px', borderTop: '1px solid var(--wb-border, #1e2a38)', display: 'flex', gap: 10, flexShrink: 0, background: 'var(--wb-bg-surface, #070a0e)' }}>

    </div>
