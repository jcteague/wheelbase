// ── P&L preview pill ──────────────────────────────────────────────────────────

// Computes CC leg P&L: (openPremium − closePrice) × contracts × 100
export function computePnl(openPremium, closePrice, contracts) {
  const open = parseFloat(openPremium)
  const close = parseFloat(closePrice)
  if (isNaN(open) || isNaN(close) || close <= 0) return null
  const pnl = (open - close) * contracts * 100
  const pct = ((open - close) / open) * 100
  return { pnl, pct }
}

export function MockPnlPreview({ openPremium, closePrice, contracts }) {
  const result = computePnl(openPremium, closePrice, contracts)
  if (!result) return null
  const { pnl, pct } = result
  const isProfit = pnl >= 0
  const color = isProfit ? 'var(--wb-green, #3fb950)' : 'var(--wb-red, #f85149)'
  const bg = isProfit ? 'rgba(63,185,80,0.07)' : 'rgba(248,81,73,0.07)'
  const border = isProfit ? 'rgba(63,185,80,0.22)' : 'rgba(248,81,73,0.22)'
  const sign = isProfit ? '+' : ''
  const pctLabel = isProfit
    ? `${Math.abs(pct).toFixed(1)}% of max`
    : `${Math.abs(pct).toFixed(1)}% above open`

  return (
    <div style={{
      display: 'flex',
      alignItems: 'center',
      justifyContent: 'space-between',
      padding: '10px 14px',
      borderRadius: 6,
      background: bg,
      border: `1px solid ${border}`,
    }}>
      <span style={{ fontSize: 11, color: 'var(--wb-text-secondary, #8899aa)', fontFamily: MONO }}>
        P&amp;L preview
      </span>
      <div style={{ display: 'flex', alignItems: 'baseline', gap: 8 }}>
        <span style={{ fontSize: 15, fontWeight: 700, color, fontFamily: MONO }}>
          {sign}{pnl < 0 ? '−' : ''}${Math.abs(pnl).toFixed(2)}
        </span>
        <span style={{ fontSize: 10, color, opacity: 0.7, fontFamily: MONO }}>
          {isProfit ? 'profit' : 'loss'} · {pctLabel}
        </span>
      </div>
    </div>
  )
}

                    value={<span style={{ color: 'var(--wb-green, #3fb950)' }}>+${pos.openPremium} / contract</span>}
                  />
                  <MockSummaryRow
                    label="Max profit (full premium)"
                    value={<span style={{ color: 'var(--wb-green, #3fb950)' }}>+$230.00</span>}
                  />
                  <MockSummaryRow
                    label="Phase transition"
                    value={
                      <span style={{ display: 'flex', alignItems: 'center', gap: 5 }}>
                        <MockPhaseBadge phase="CC_OPEN" />
                        <span style={{ fontSize: 9, color: 'var(--wb-text-muted, #4a5a6a)' }}>→</span>
                        <MockPhaseBadge phase="HOLDING_SHARES" />
                      </span>
                    }
                  />
                  <MockSummaryRow
                    label="Cost basis after close"
                    value={`$${pos.costBasisPerShare} / share (unchanged)`}
                    noBorder
                    highlight
                  />
                </MockSectionCard>

                {/* Close price field
                  Real:
                    <Field label="Close Price (Buy to Close)" error={priceError} hint="Enter the fill price per contract">
                      <Input type="number" … />
                    </Field>
                */}
                <MockField
                  label="Close Price (Buy to Close)"
                  error={priceError}
                  hint={!priceError ? 'Enter the fill price per contract' : undefined}
                >
                  <MockPriceInput
                    value={activePriceStr}
                    hasError={Boolean(priceError)}
                    onChange={setClosePrice}
                    placeholder="0.00"
                  />
                </MockField>

                {/* Live P&L preview — updates as close price changes
                  Real: computed inline from RHF watch() + shown below the price field */}
                {!priceError && (
                  <MockPnlPreview
                    openPremium={pos.openPremium}
                    closePrice={isFormLoss ? '3.50' : activePriceStr}
                    contracts={pos.contracts}
                  />
                )}

                {/* Fill date field
                  Real:
                    <Field label="Fill Date" error={dateError} hint="The date your broker executed the buy-to-close">
                      <DatePicker … />
                    </Field>
                */}
                <MockField
                  label="Fill Date"
                  error={dateErrorMsg}
                  hint={!dateErrorMsg ? 'The date your broker executed the buy-to-close' : undefined}
                >
                  <MockDatePicker
                    value={activeDateStr}
                    hasError={Boolean(dateErrorMsg)}
                    onChange={setFillDate}
                  />
                </MockField>

                {/* Irrevocable warning
                  Real: <AlertBox variant="warning"><strong>This cannot be undone.</strong>…</AlertBox> */}
                <MockAlertBox variant="warning">
                  <strong style={{ fontWeight: 700, display: 'block', marginBottom: 3 }}>This cannot be undone.</strong>
                  A CC_CLOSE leg will be recorded. The position returns to Holding Shares. Full leg history is preserved.
                </MockAlertBox>
              </>
            )}
          </div>

          {/* ── Footer ── */}
          {!isSuccess && (
            <div style={{
              padding: '16px 24px',
              borderTop: '1px solid var(--wb-border, #1e2a38)',
              display: 'flex',
              gap: 10,
              flexShrink: 0,
              background: 'var(--wb-bg-surface, #070a0e)',
            }}>
              {/* Real: <Button variant="outline" onClick={onClose} style={{ flex: 1 }}>Cancel</Button> */}
              <MockOutlineButton style={{ flex: 1 }}>Cancel</MockOutlineButton>
              {/* Real: <FormButton label="Confirm Close" pendingLabel="Closing…" isPending={isPending} style={{ flex: 1 }} /> */}
              <MockFormButton
                label="Confirm Close"
                onClick={() => {
                  const price = isFormLoss ? '3.50' : activePriceStr
                  const pnl = computePnl(pos.openPremium, price, pos.contracts)
                  if (pnl && pnl.pnl >= 0) setState('success-profit')
                  else setState('success-loss')
                }}
                style={{ flex: 1, fontWeight: 700 }}
              />
            </div>
          )}

        </div>
      </div>
    </div>
  )
}
