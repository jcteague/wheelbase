# US-12/13: Roll CSP Form

_Roll an open CSP to a new expiration (roll out) or a different strike and expiration (roll down and out), with live net credit/debit preview._

## Acceptance Criteria States

- **credit**: Roll out with net credit — $180 strike unchanged, new May expiration, cost to close $1.20, new premium $2.80, net credit $1.60/contract
- **debit-warning**: Roll with net debit — cost to close $3.00, new premium $2.50, net debit $0.50/contract shown in amber with warning
- **validation-error**: New expiration before current expiration — red inline error
- **down-and-out**: Roll down and out — strike changed from $180 to $175, label shows "Roll Down & Out"
- **success**: Roll confirmed — linked ROLL_FROM/ROLL_TO pair created, position stays CSP_OPEN

export function getRollTypeLabel(currentStrike, newStrike, currentExp, newExp) {
  const sameStrike = currentStrike === newStrike
  const strikeUp = parseFloat(newStrike) > parseFloat(currentStrike)
  const strikeDown = parseFloat(newStrike) < parseFloat(currentStrike)
  if (sameStrike) return 'Roll Out'
  if (strikeDown) return 'Roll Down & Out'
  if (strikeUp) return 'Roll Up & Out'
  return 'Roll'
}

export function NetCreditDebitPreview({ costToClose, newPremium, contracts }) {
  const close = parseFloat(costToClose)
  const prem = parseFloat(newPremium)
  if (isNaN(close) || isNaN(prem) || close <= 0 || prem <= 0) return null
  const net = prem - close
  const isCredit = net >= 0
  const total = Math.abs(net) * contracts * 100
  const color = isCredit ? 'var(--wb-green, #3fb950)' : 'var(--wb-gold, #e6a817)'
  const bg = isCredit ? 'rgba(63,185,80,0.07)' : 'rgba(230,168,23,0.07)'
  const border = isCredit ? 'rgba(63,185,80,0.22)' : 'rgba(230,168,23,0.22)'
  const label = isCredit ? 'Net Credit' : 'Net Debit'
  const sign = isCredit ? '+' : '-'
  return (
    <div style={{ borderRadius: 6, background: bg, border: '1px solid ' + border, overflow: 'hidden' }}>
      <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between', padding: '12px 14px' }}>
        <span style={{ fontSize: 11, color: 'var(--wb-text-secondary, #8899aa)', fontFamily: MONO }}>{label}</span>
        <div style={{ display: 'flex', alignItems: 'baseline', gap: 8 }}>
          <span style={{ fontSize: 17, fontWeight: 700, color, fontFamily: MONO }}>{sign}${Math.abs(net).toFixed(2)}/contract</span>
          <span style={{ fontSize: 11, color, opacity: 0.7, fontFamily: MONO }}>(${total.toFixed(2)} total)</span>
        </div>
      </div>
      {!isCredit && (
        <div style={{ padding: '8px 14px', borderTop: '1px solid ' + border, fontSize: 11, color, opacity: 0.85, fontFamily: MONO }}>
          ⚠ This roll costs more to close than the new premium provides
        </div>
      )}
    </div>
  )
}
